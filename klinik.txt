<script type="module">
import { initializeApp } from "https://www.gstatic.com/firebasejs/12.12.0/firebase-app.js";
import { getDatabase, ref, push, onValue, update, remove, get }
from "https://www.gstatic.com/firebasejs/12.12.0/firebase-database.js";

const app = initializeApp({
  apiKey: "AIzaSyD9t_MExZTAsNIKwTjZSs8sLZ2FWyV7YBQ",
  databaseURL: "https://antrean-klinik-fd95a-default-rtdb.asia-southeast1.firebasedatabase.app"
});

const db = getDatabase(app);

// 🔊 FIX SUARA (WAJIB)
document.body.addEventListener("click", () => {
  speechSynthesis.resume();
});

// 🔄 RESET HARIAN
const todayKey = new Date().toDateString();
const tRef = ref(db,"today");

const tSnap = await get(tRef);
if(!tSnap.exists() || tSnap.val() !== todayKey){
  await remove(ref(db,"pasien"));
  await update(tRef, todayKey);
}

// ⏰ JAM 22
const allow = new Date().getHours() >= 22;

// DATA
const pasienRef = ref(db,"pasien");
const currentRef = ref(db,"current");

let myId = localStorage.getItem("id");

// CURRENT
onValue(currentRef,s=>{
 document.getElementById("current").innerText = s.val() || "-";
});

// 🔢 NOMOR ANTI BUG
async function getNomor(){
 const snap = await get(pasienRef);

 if(!snap.exists()) return "A001";

 let max = 0;
 Object.values(snap.val()).forEach(d=>{
   let n = parseInt(d.nomor.replace("A",""));
   if(n > max) max = n;
 });

 let next = max + 1;
 return "A" + String(next).padStart(3,"0");
}

// AMBIL
window.ambil = async()=>{
 if(!allow) return alert("Buka jam 22:00");

 const nama = document.getElementById("nama").value;
 if(!nama) return alert("Isi nama");

 const nomor = await getNomor();
 const r = push(pasienRef);

 await update(r,{
  nama,
  nomor,
  pesan: document.getElementById("pesan").value || "-",
  status:"menunggu",
  waktu:Date.now()
 });

 myId = r.key;
 localStorage.setItem("id",myId);

 listenMe();
};

// STATUS REALTIME (FIX)
function listenMe(){
 if(!myId) return;

 onValue(ref(db,"pasien/"+myId), s=>{
  if(!s.exists()){
    document.getElementById("me").innerText = "Sudah dipanggil / tidak ada";
    return;
  }

  const d = s.val();
  document.getElementById("me").innerHTML =
    `<b>${d.nomor}</b><br>${d.nama}<br>${d.status}`;
 });
}

// CHECKIN
window.checkin = ()=>{
 if(!myId) return;
 update(ref(db,"pasien/"+myId),{status:"checkin"});
};

// BATAL
window.batal = ()=>{
 if(!myId) return;
 remove(ref(db,"pasien/"+myId));
 localStorage.removeItem("id");
 document.getElementById("me").innerText="Dibatalkan";
};

// LOGIN
window.login = ()=>{
 if(document.getElementById("pass").value === "admin123"){
  document.getElementById("admin").style.display="block";
  loadAdmin();
 }else alert("Salah");
};

// ADMIN LIST (FIX)
function loadAdmin(){
 onValue(pasienRef,s=>{
  const el = document.getElementById("admin");
  el.innerHTML="";

  if(!s.exists()) return;

  let arr=[];
  s.forEach(c=>arr.push([c.key,c.val()]));

  arr.sort((a,b)=>a[1].waktu - b[1].waktu);

  arr.forEach(([id,d])=>{
   el.innerHTML += `
   <div>
   ${d.nomor} - ${d.nama} (${d.status})<br>
   <button onclick="panggil('${id}','${d.nama}','${d.nomor}')">Panggil</button>
   </div>`;
  });
 });
}

// 🔊 PANGGIL (FIX TOTAL)
window.panggil = (id,nama,nomor)=>{
 update(currentRef, nomor+" - "+nama);

 setTimeout(()=>{
   remove(ref(db,"pasien/"+id));
 },300);

 const suara = new SpeechSynthesisUtterance(
  `Nomor antrean ${nomor}, atas nama ${nama}, silakan masuk`
 );

 suara.lang = "id-ID";
 speechSynthesis.speak(suara);
};

// AUTO LOAD
if(myId) listenMe();
